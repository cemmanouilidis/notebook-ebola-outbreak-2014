def cell_text(cell):
    return " ".join(cell.stripped_strings)

def get_table_columns(table):
    headers = []
    
    # get row containing the table headers
    row = table.find("tr")
    
    # skip first header "Date of report" this is the index for the DataFrame
    # squash sub headers into remaining main headers
    subheaders = ("cases", "deaths")
    for h in row.find_all("th")[1:]:
        for sh in subheaders:
            headers.append("%s %s" % (cell_text(h), sh))
     
    return headers

def get_table_data_and_index(table):
    data = []
    index = []
    
    # extract data for each row beginning with row #3
    # row #1 contains the headers 
    # row #2 contains subheaders "cases" and "deaths"
    for row in table.findAll('tr')[2:]:        
        rd = list(map(cell_text, row.find_all(re.compile('t[dh]') ))) 
        # row data for column "Date of report" go to the index
        index.append(rd[0])
        # remaining row data go into list data
        assert(len(rd[1:]) == 10)
        data.append(rd[1:])
        
    return (data, pd.DatetimeIndex(index))
